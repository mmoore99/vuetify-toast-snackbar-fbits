Vuetify Toast Snackbar Demo

Show 'hello' Show Custom Show 'error' Show toast for 10 sec Clear queue 

          this.$toast.success('Hello', {
            queueable: true
          });

          this.$toast.error('Error');

          this.$toast('This is an information.', {
            timeout: 10000,
            icon: 'info',
            dismissable: false,
            showClose: true
          });
